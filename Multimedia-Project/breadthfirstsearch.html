Breadth First Search

bool visit[105];

bool found=false;

queue<int>que;

int value=6;

visit[value]=true;

while(que.empty()==false){

     int currentValue=que.front();

     que.pop();

     if(currentValue==value){

          found=true;

          break;

     }

     for(int i=0;i<adj[currentValue].size();i++){

          if(visit[adj[currentValue][i]]==false){

               visit[adj[currentValue][i]]=true;

               que.push(adj[currentValue][i]);

          }

     }

}

if(found) printf("Found\n");

else printf("Not found\n");


>Breadth First Search is graph traversal search that prioritse neighbor nodes first at each depth

>Complexity O(|V|)
